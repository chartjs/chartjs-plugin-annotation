# Yearly quarters

const config = {
  type: 'line',
  data,
  options: {
    scales: {
      y: {
        beginAtZero: true,
        max: 120,
        min: 0
      }
    },
    plugins: {
      annotation: {
        drawTime: 'beforeDraw',
        annotations: {
          annotation1,
          annotation2,
          annotation3,
          annotation4
        }
      }
    }
  }
};
const annotation1 = {
  type: 'box',
  backgroundColor: 'rgba(255,245,157,0.2)',
  borderWidth: 0,
  xMax: 2.5,
  xMin: -0.5,
  label: {
    drawTime: 'afterDraw',
    enabled: true,
    content: 'First quarter',
    position: {
      x: 'center',
      y: 'start'
    }
  }
};
