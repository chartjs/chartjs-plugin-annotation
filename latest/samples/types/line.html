# Line

const config = {
  type: 'line',
  data,
  options: {
    plugins: {
      annotation: {
        annotations: {
          annotation1,
          annotation2
        }
      }
    },
    // Core options
    scales: {
      y: {
        stacked: true
      }
    }
  }
};
const annotation1 = {
  type: 'line',
  scaleID: 'y',
  borderWidth: 3,
  borderColor: 'black',
  value: minValue,
  label: {
    content: (ctx) => 'Lower bound: ' + minValue(ctx).toFixed(3),
    enabled: true
  },
};
const annotation2 = {
  type: 'line',
  scaleID: 'y',
  borderWidth: 3,
  borderColor: 'black',
  value: maxValue,
  label: {
    rotation: 'auto',
    backgroundColor: 'black',
    borderColor: 'red',
    borderDash: [6, 3],
    borderRadius: 10,
    borderWidth: 2,
    content: (ctx) => 'Upper bound: ' + maxValue(ctx).toFixed(3),
    enabled: true
  }
};
function minValue(ctx) {
  const dataset = ctx.chart.data.datasets[0];
  const min = dataset.data.reduce((max, point) => Math.min(point, max), Infinity);
  return isFinite(min) ? min : 0;
}
function maxValue(ctx) {
  const datasets = ctx.chart.data.datasets;
  const count = datasets[0].data.length;
  let max = 0;
  for (let i = 0; i < count; i++) {
    let sum = 0;
    for (const dataset of datasets) {
      sum += dataset.data[i];
    }
    max = Math.max(max, sum);
  }
  return max;
}
